\begin{tcolorbox}[hbox, title=Derivative rules]
  \begin{minipage}{0.5\textwidth}
    \begin{description}
      \item[Sum]
      \[
        (f(x)+g(x))' = f'(x) + g'(x)
      \]
      \item[Difference]
      \[
        (f-g)'(x) = f'(x) - g'(x)
      \]
      \item[Product]
      \[
        (fg)'(x) = f'(x)g(x) + f(x)g'(x)
      \]
      \item[Sum]
      \[
        \left(\frac{f(x)}{g(x)}\right)' = \frac{f'(x)g(x) - f(x)g'(x)}{g(x)^2}
      \]
      \item[Chain rule]
      \[
        (f(g))'(x) = f'(g(x))g'(x)
      \]
      \item[Inverse]
      \[
        f^{-1}'(x) = \frac{1}{f'(x)}
      \]
    \end{description}
  \end{minipage}
\end{tcolorbox}
